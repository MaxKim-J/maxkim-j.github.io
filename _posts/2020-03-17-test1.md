## 참 재밌는 개발

오늘은 개발을 해따 내일도 개발을 할거다 아마 모레도..  
개발하다 죽자! 죽쟈! 죽쟈!

{% highlight javascript %}

// 맵객체 set, 인수로 첫번째는 키, 두번째는 값
filters.set("견종", "레브라도레트리버");

// 맵객체 get, 인수로는 키만 전달
filters.get("견종");

// 이거랑
let filters = new Map()
 .set("견종", "레브라도리트리버")
 .set("크기", "대형견")
 .set("색상", "갈색");

filters.get("크기");

// 요거랑 같다
let filters = new Map([
  ["견종", "레브라도리트리버"],
  ["크기", "대형견"],
  ["색상", "갈색"]
]);

filters.get("색상");

// 심지어 이터레이터이기도 함
function checkFilters(filters) {
  for (const entry of filters) {
    console.log(entry);
  }
}

{% endhighlight %}
